let allGroupedData = []; let filteredData = []; let currentIndex = 0; async function init() { try { // Fetching with cache-busting and error checking const response = await fetch('data.json?v=' + Date.now(), { cache: "no-store" }); if (!response.ok) throw new Error("HTTP Status " + response.status); const rawData = await response.json(); const groups = {}; rawData.forEach(e => { if (!e.markets || e.markets.length === 0) return; const title = e.title.replace(/by ___ \?/g, "").trim(); let cat = 'gen'; const t = title.toLowerCase(); if (t.includes('trump') || t.includes('election') || t.includes('biden') || t.includes('senate')) cat = 'us'; else if (t.includes('israel') || t.includes('ukraine') || t.includes('china') || t.includes('parliament') || t.includes('un ')) cat = 'world'; else if (t.includes('nfl') || t.includes('nba') || t.includes('soccer') || t.includes('fifa')) cat = 'sports'; e.markets.forEach(m => { let p = 0; try { p = JSON.parse(m.outcomePrices)[0]; } catch(err) { p = m.price || 0; } const prob = Math.round(parseFloat(p) * 100); if (prob < 5 || prob > 95) return; if (!groups[title]) { groups[title] = { title, slug: e.slug, vol: parseFloat(e.volume || 0), vol24: parseFloat(e.volume24hr || 0), cat, markets: [] }; } groups[title].markets.push({ prob, choice: m.groupItemTitle || "Yes" }); }); }); const sorted = Object.values(groups).sort((a,b) => b.vol24 - a.vol24); let counts = { us: 0, world: 0, sports: 0 }; let final = []; sorted.forEach(m => { const polTotal = counts.us + counts.world; if (m.cat === 'us' && counts.us < 2 && polTotal < 4) { final.push(m); counts.us++; } else if (m.cat === 'world' && counts.world < 2 && polTotal < 4) { final.push(m); counts.world++; } else if (m.cat === 'sports' && counts.sports < 3) { final.push(m); counts.sports++; } else if (m.cat === 'gen' || final.length < 10) { if (!final.find(x => x.title === m.title)) final.push(m); } }); allGroupedData = final.slice(0, 50); document.getElementById('status').innerText = "System Live"; render(); } catch (e) { console.error("Pulse Error:", e); document.getElementById('status').innerText = "Sync Offline - 404/Network Error"; } } function formatV(n) { return n >= 1e6 ? (n/1e6).toFixed(1)+'M' : n >= 1e3 ? (n/1e3).toFixed(1)+'k' : n; } function loadNext() { const list = document.getElementById('odds-list'); const batch = filteredData.slice(currentIndex, currentIndex + 10); batch.forEach((g, i) => { g.markets.sort((a,b) => b.prob - a.prob); const main = g.markets[0]; const id = `g-${currentIndex + i}`; const trending = (currentIndex + i < 3); list.insertAdjacentHTML('beforeend', `
${trending ? '' : ''} ${main.prob}%
${g.title}${main.choice}
$${formatV(g.vol)} Wanna Bet?
${g.markets.slice(1).map(m => `
${m.prob}% ${m.choice} Wanna Bet?
`).join('')}
`); }); currentIndex += 10; document.getElementById('load-more-btn').style.display = currentIndex >= filteredData.length ? "none" : "block"; } function render() { filteredData = [...allGroupedData]; document.getElementById('odds-list').innerHTML = ""; currentIndex = 0; loadNext(); } function handleSearch() { const q = document.getElementById('search-input').value.toLowerCase(); filteredData = allGroupedData.filter(m => m.title.toLowerCase().includes(q)); document.getElementById('odds-list').innerHTML = ""; currentIndex = 0; loadNext(); } function scrollToTop() { window.scrollTo({top: 0, behavior: 'smooth'}); } window.toggleGroup = id => { const e = document.getElementById(id); e.style.display = e.style.display === "block" ? "none" : "block"; }; window.onscroll = () => document.getElementById('back-to-top').style.display = (window.scrollY > 300) ? "flex" : "none"; init();
